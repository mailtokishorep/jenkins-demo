    agent {
      label 'Linux Node'
    }
   
   
   
    stages {

  steps {
    sh 'docker version'
  }
}

  steps {
    sh '''docker image ls 
      docker image build .  -f Dockerfile -t kulbhushanmayer/devops:latest
      docker image ls'''
  }
}
        
        stage('How are you?') {
            steps {
                echo 'How are you?'
            }
        }
        stage ('SAST'){
          parallel{
            stage ('sonar-scan'){
              steps {
                echo 'Running Sonar Scan'
              }
            }
            stage ('synk-scan'){
              steps {
                echo 'Synk scan'
              }
            }
            stage ('DC'){
              steps {
                echo 'Running Dependency Check'
              }
            }
          }
        }
    }
}
